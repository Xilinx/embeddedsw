cmake_minimum_required(VERSION 3.15)
project(scuwdt)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xscuwdt.c)
collect (PROJECT_LIB_HEADERS xscuwdt.h)
collect (PROJECT_LIB_SOURCES xscuwdt_g.c)
collect (PROJECT_LIB_HEADERS xscuwdt_hw.h)
collect (PROJECT_LIB_SOURCES xscuwdt_selftest.c)
collect (PROJECT_LIB_SOURCES xscuwdt_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(scuwdt STATIC ${_sources})
set_target_properties(scuwdt PROPERTIES LINKER_LANGUAGE C)
