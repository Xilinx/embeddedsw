cmake_minimum_required(VERSION 3.15)
project(rtcpsu)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xrtcpsu_sinit.c)
collect (PROJECT_LIB_SOURCES xrtcpsu_g.c)
collect (PROJECT_LIB_SOURCES xrtcpsu_selftest.c)
collect (PROJECT_LIB_SOURCES xrtcpsu.c)
collect (PROJECT_LIB_SOURCES xrtcpsu_intr.c)
collect (PROJECT_LIB_HEADERS xrtcpsu.h)
collect (PROJECT_LIB_HEADERS xrtcpsu_hw.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(rtcpsu STATIC ${_sources})
set_target_properties(rtcpsu PROPERTIES LINKER_LANGUAGE C)
