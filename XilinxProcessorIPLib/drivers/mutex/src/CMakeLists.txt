cmake_minimum_required(VERSION 3.15)
project(mutex)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xmutex.c)
collect (PROJECT_LIB_HEADERS xmutex.h)
collect (PROJECT_LIB_SOURCES xmutex_g.c)
collect (PROJECT_LIB_HEADERS xmutex_hw.h)
collect (PROJECT_LIB_SOURCES xmutex_selftest.c)
collect (PROJECT_LIB_SOURCES xmutex_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(mutex STATIC ${_sources})
set_target_properties(mutex PROPERTIES LINKER_LANGUAGE C)
