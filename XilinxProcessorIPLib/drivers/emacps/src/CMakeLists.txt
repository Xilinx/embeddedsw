cmake_minimum_required(VERSION 3.15)
project(emacps)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xemacps_g.c)
collect (PROJECT_LIB_SOURCES xemacps_hw.c)
collect (PROJECT_LIB_SOURCES xemacps_control.c)
collect (PROJECT_LIB_SOURCES xemacps.c)
collect (PROJECT_LIB_SOURCES xemacps_bdring.c)
collect (PROJECT_LIB_HEADERS xemacps_bd.h)
collect (PROJECT_LIB_SOURCES xemacps_sinit.c)
collect (PROJECT_LIB_HEADERS xemacps_hw.h)
collect (PROJECT_LIB_SOURCES xemacps_intr.c)
collect (PROJECT_LIB_SOURCES xemacps_pcs.c)
collect (PROJECT_LIB_HEADERS xemacps.h)
collect (PROJECT_LIB_HEADERS xemacps_bdring.h)
collect (PROJECT_LIB_HEADERS xemacps_pcs.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(emacps STATIC ${_sources})
set_target_properties(emacps PROPERTIES LINKER_LANGUAGE C)
