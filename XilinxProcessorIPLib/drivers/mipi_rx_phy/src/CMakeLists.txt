cmake_minimum_required(VERSION 3.14.7)
project(mipi_rx_phy)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xmipi_rx_phy.c)
collect (PROJECT_LIB_HEADERS xmipi_rx_phy.h)
collect (PROJECT_LIB_SOURCES xmipi_rx_phy_g.c)
collect (PROJECT_LIB_HEADERS xmipi_rx_phy_hw.h)
collect (PROJECT_LIB_SOURCES xmipi_rx_phy_selftest.c)
collect (PROJECT_LIB_SOURCES xmipi_rx_phy_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(mipi_rx_phy STATIC ${_sources})
set_target_properties(mipi_rx_phy PROPERTIES LINKER_LANGUAGE C)
