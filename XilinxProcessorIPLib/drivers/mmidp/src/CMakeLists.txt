cmake_minimum_required(VERSION 3.15)
project(mmidp)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_HEADERS xmmidp.h)
collect (PROJECT_LIB_HEADERS xmmidp_hw.h)
collect (PROJECT_LIB_SOURCES xmmidp_phy.c)
collect (PROJECT_LIB_SOURCES xmmidp_intr.c)
collect (PROJECT_LIB_SOURCES xmmidp_link.c)
collect (PROJECT_LIB_SOURCES xmmidp_stream.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(mmidp STATIC ${_sources})
set_target_properties(mmidp PROPERTIES LINKER_LANGUAGE C)
