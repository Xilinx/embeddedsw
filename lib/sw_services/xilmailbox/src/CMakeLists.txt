cmake_minimum_required(VERSION 3.14.7)
#include(${CMAKE_CURRENT_SOURCE_DIR}/../../../../cmake/common.cmake NO_POLICY_SCOPE)
include(${CMAKE_CURRENT_SOURCE_DIR}/XilmailboxExample.cmake NO_POLICY_SCOPE)
project(xilmailbox)

find_package(common)
include(${CMAKE_CURRENT_SOURCE_DIR}/xilmailbox.cmake NO_POLICY_SCOPE)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/PS/)
collect (PROJECT_LIB_SOURCES xilmailbox.c)
collect (PROJECT_LIB_HEADERS xilmailbox.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
if (${NON_YOCTO})
file(COPY ${CMAKE_BINARY_DIR}/include/xilmailbox_hwconfig.h DESTINATION ${CMAKE_INCLUDE_PATH}/)
endif()
add_library(xilmailbox STATIC ${_sources})
set_target_properties(xilmailbox PROPERTIES LINKER_LANGUAGE C)
get_headers(${_headers})
set_target_properties(xilmailbox PROPERTIES ADDITIONAL_CLEAN_FILES "${CMAKE_LIBRARY_PATH}/libxilmailbox.a;${CMAKE_INCLUDE_PATH}/xilmailbox_hwconfig.h;${clean_headers}")
install(TARGETS xilmailbox LIBRARY DESTINATION ${CMAKE_LIBRARY_PATH} ARCHIVE DESTINATION ${CMAKE_LIBRARY_PATH})
install(DIRECTORY ${CMAKE_BINARY_DIR}/include DESTINATION ${CMAKE_INCLUDE_PATH}/..)
