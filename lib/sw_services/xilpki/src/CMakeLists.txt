cmake_minimum_required(VERSION 3.15)
project(xilpki)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xilpki.c)
collect (PROJECT_LIB_HEADERS xilpki.h)
collect (PROJECT_LIB_HEADERS xilpki_ecdsa.h)
collect (PROJECT_LIB_SOURCES xilpki_kat.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(xilpki STATIC ${_sources})
set_target_properties(xilpki PROPERTIES LINKER_LANGUAGE C)
get_headers(${_headers})
set_target_properties(xilpki PROPERTIES ADDITIONAL_CLEAN_FILES "${CMAKE_LIBRARY_PATH}/libxilpki.a;${clean_headers}")
install(TARGETS xilpki LIBRARY DESTINATION ${CMAKE_LIBRARY_PATH} ARCHIVE DESTINATION ${CMAKE_LIBRARY_PATH})
install(DIRECTORY ${CMAKE_BINARY_DIR}/include DESTINATION ${CMAKE_INCLUDE_PATH}/..)
